def bp_colnew(kappa,eta,gamma,mu=0.,eps=10.,lam=0.,beta=0.,delta=0.,sigmaS=0.,sigmaO=0.,DoPlot=False):
    # out,sol=bp_colnew(10,.1,.2,DoPlot=True)

    # print 'kappa,eta,gamma,mu,eps,lam,beta,delta'
    # print(kappa,eta,gamma,mu,eps,lam,beta,delta)
          
    ae = 1/(1+exp(-eps))            # Dimensionless potential boundary condition at the electrode surface from the Nernst eqn

    def Ra(a,s,o): 
        num= kappa**2*a*s
        denom= gamma*a*(1. + mu*s)+s*(1.+o*lam)+beta*o
        return num/denom  * (a >= 0) * (s >= 0)
    
    def Ro(a,s,o):
        num= kappa**2 * beta/eta * delta * o * s
        denom= gamma*a*(1. + mu*s)+s*(1.+o*lam)+beta*o
        return num/denom  * (o >= 0) * (s >= 0)
    
    
    degrees = [1, 1, 1, 1, 1, 1] 
    
    def deq(x,z):  # z=[a(x) va(x) s(x) vs(x), o(x), vo(x)];

        a, va, s, vs, o, vo = z

        dvadx= Ra(a,s,o)
        dadx= va

        dvsdx= gamma/eta*Ra(a,s,o) + 1/delta * Ro(a,s,o)
        dsdx= vs

        dvodx= Ro(a,s,o)
        dodx= vo
        
        dzdx= array([dadx,dvadx,dsdx,dvsdx,dodx,dvodx]); return dzdx

    boundary_points = [0, 0, 0, 1, 1, 1]

    def bc(z):

        a, va, s, vs, o, vo = z

        res=   array([  a[0] - 1/(1+exp(-eps)),     # Nernst BC on a at electrode-film interface
                       vs[1],                       # Zero flux BC on s at electrode-film interface
                       vo[2],                       # Zero flux BC on o at electrode-film interface
                       va[3],                       # Zero flux BC on a at film-solution interface
                        s[4] - 1 + sigmaS*vs[4],    # Mass transfer BC on s at film-solution interface
                        o[5] - 1 + sigmaO*vo[5] ] ) # Mass transfer BC on o at film-solution interface

        return res

    def guess(x):
        v = ones_like(x)
        dv= zeros_like(x)
        u= array([v, dv, v, dv, v, dv])
        du = array([dv, dv, dv, dv, dv, dv])
        return u, du

    class struct:
        pass
    
    tol = [1e-6, 1e-6, 1e-6, 1e-6, 1e-6, 1e-6]


    try:
       solution = bvp.colnew.solve(boundary_points, degrees, deq, bc,
                        tolerances=tol,
                        vectorized=True, 
                        maximum_mesh_size=300, 
                        initial_guess=guess)
    except:
        Jobs=nan
        
        class solutionerror:
            pass

        solution=solutionerror()
        
    else:
        x=solution.mesh
        y = solution(x)
        a = y[:,0]
        s = y[:,2]
        R = kappa**2*a*s/(gamma*a*(1+mu*s)+s) * (a >= 0) * (s >= 0)

        Jobs=-y[0,1]

        v=struct()
        x=solution.mesh
        y = solution(x)
        a = y[:,0]
        s = y[:,2]
        o = y[:,4]
        ra = Ra(a,s,o)
        ro = Ro(a,s,o)
        rs = gamma/eta*Ra(a,s,o) + 1/delta * Ro(a,s,o)

        v.x, v.y, v.a, v.s, v.o, v.Ra, v.Rs, v.Ro= (x,y,a,s,o,ra,rs,ro)
        solution.v=v

        if DoPlot:
            solplot(solution)

    nd=struct()
    
    nd.kappa, nd.eta, nd.gamma, nd.mu, nd.eps, nd.lam, nd.beta, nd.delta, nd.sigmaS, nd.sigmaO = \
    (  kappa,    eta,    gamma,    mu,    eps,    lam,    beta,    delta,    sigmaS,    sigmaO  )
    
    solution.nd=nd
                
    return Jobs, solution

def solplot(sol):
    x= sol.mesh
    y= sol(x)
    plot(x,y[:,[0,2,4]])
    ylim(0,1.1)
    plot([0, 1], [1, 1], color='grey', linestyle=':', linewidth=0.5)
    xlabel('x / $\ell$')
    legend(['a','s','o'],loc='best')

def bp(V,ST,kcat,KM,KS,OT=0.,KI=1e6,kO=0.):
    
    kM=kcat/KM
    
    kappa= L*sqrt(vm*kM*ET/DM)
    eta=   (vm*DS*kM*KS)/(vs*DM*kcat)
    gamma= (kM*Mtot*KS)/(kcat*ST)
    mu=    ST/KS
    sigmaS= pS*dS/L
    
    eps = (n * F * (V-U))/(R*T)    # From the Nernst equation

    lam=   OT/(pO*KI)
    beta=  KS*kO*OT/kcat/ST
    sigmaO= dO/L
    
    if OT==0:
        delta=1
    else:
        delta= DS*ST/DO/OT    

    Jobs, sol= bp_colnew(kappa,eta,gamma,mu,eps,lam,beta,delta,sigmaS,sigmaO)
    
    icd= n*F/vm*DM*Mtot/L*Jobs*1000   # mA/cm2
    
    sol.imax= n*F*kcat*ET*L*1000      # mA/cm2
    return icd, sol

bpv=vectorize(bp)

def bpdf(df,kcat,KM,KS,KI=1.0e6,kO=0.):
    
    ic=zeros_like(df['E'],dtype='float')
    
    V=df['E'].values
    S=df['S'].values
    O=pO*df['O'].values

    
    for ii,Vval in enumerate(V):
        ic[ii],sol=bp(Vval,S[ii],kcat,KM,KS,O[ii],KI,kO)
    
    return ic,sol

def make_fitfun(df):
    def fitfun(ix,kcat,KM,KS,KI=1.0e6,kO=0.):

        icf,sol=bpdf(df.ix[ix],kcat,KM,KS,KI,kO)

        return icf

    return fitfun

def fitreport(yf,yx,popt,pcov):
    
    perr=sqrt(diag(pcov))
    labels=['kcat','KM','KS','KI','kO']
    
    # for ii,p in enumerate(popt):
    #   print labels[ii],'=',p,u'',perr[ii] 
        
    for ii,p in enumerate(popt):
        print(u'%s = %7.3g  %7.3g') % (labels[ii],p,perr[ii]) 
    
    kM= popt[0]/popt[1]
    kS= popt[0]/popt[2]

    print(u'%s = %7.3g') % ('kM',kM) 
    print(u'%s = %7.3g') % ('kS',kS) 
        
    SSres= sum((yx-yf)**2)
    SStot= sum((yx-mean(yx))**2)
    R2=1-SSres/SStot
    print('R2 = %7.4g') % R2
